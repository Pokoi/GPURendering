160  void input()
161  {
162  sf::Event e;
163  while (window.poll_event(e))
164  {
165  switch (e.type)
166  {
167  case sf::Event::Closed:
168  is_running = false;
169  break;
170 
171  case sf::Event::KeyPressed:
172  switch (e.key.code)
173  {
174  case sf::Keyboard::Escape: is_running = false; break;
175  case sf::Keyboard::W: camera.set_y_displacement(+1); break;
176  case sf::Keyboard::S: camera.set_y_displacement(-1); break;
177  case sf::Keyboard::A: camera.set_x_displacement(-1); break;
178  case sf::Keyboard::D: camera.set_x_displacement(+1); break;
179  case sf::Keyboard::E: view.toggle_postproccess(); break;
180  }
181  break;
182 
183  case sf::Event::KeyReleased:
184  {
185 
186  switch (e.key.code)
187  {
188  case sf::Keyboard::W: camera.set_y_displacement(0); break;
189  case sf::Keyboard::S: camera.set_y_displacement(0); break;
190  case sf::Keyboard::A: camera.set_x_displacement(0); break;
191  case sf::Keyboard::D: camera.set_y_displacement(0); break;

193  }

197  }
